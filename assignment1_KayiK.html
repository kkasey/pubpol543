link='https://github.com/EvansDataScience/data/raw/master/eduwa.rda'

#getting the data TABLE from the file in the cloud:
load(file=url(link))
CountyCount_ST=aggregate(data=eduwa,
                         Student.Teacher.Ratio~County,
                         FUN=mean)
CountyCount_ST=CountyCount_ST[order(-CountyCount_ST$Student.Teacher.Ratio),]
head(CountyCount_ST,20)
library(ggplot2)
base1=ggplot(data=CountyCount_ST, 
             aes(x=County,Student.Teacher.Ratio)) + theme_classic()
base2=base1+scale_x_discrete()
bar1=base2  +geom_bar(stat = "identity", fill="gray")   
bar2=bar1 + coord_flip() 
bar2

base2=base1+scale_x_discrete(limits=CountyCount_ST$County)
bar1=base2  +geom_bar(stat = "identity", fill="gray")
bar2=bar1 + coord_flip() 
bar2

# Average student-teacher ratio in WA schools

bar3 = bar2 + geom_vline(xintercept = 15 ,
                         linetype="dashed", size=1, color='black')
bar3
